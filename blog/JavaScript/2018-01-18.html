# 反转字符串

function reverseString (str) {
  return str.split('').reverse().join('');
}

// Version ≥ 6
function reverseString (str) {
  return [...String(str)].reverse().join('');
}

console.log(reverseString('stackoverflow')); // "wolfrevokcats"
console.log(reverseString(1337)); // "7331"
console.log(reverseString([1, 2, 3])); // "3,2,1"

# 判断字符串类型

var isString = function (value) {
  return typeof value === "string" || value instanceof String;
};
var aString = "Primitive String";
var aStringObj = new String("String Object");
isString(aString); // true
isString(aStringObj); // true
isString({}); // false
isString(5); // false

Object.prototype.toString.call(aString); //"[object String]"
Object.prototype.toString.call(aStringObj); //"[object String]"
// 推荐这种方法，因为它还可以用来检测其他类型

// 通过检测是否含有需要的方法也可以用来判断是不是字符串
// Generic check for a substring method
if (aString.substring) {}
// Explicit check for the String substring prototype method
if (aString.substring === String.prototype.substring) {}

# 小数的进制转换

let b16 = '3.243f3e0370cdc';
// Split into integer and fraction parts
let [i16, f16] = b16.split('.');
// Calculate base 10 integer part
let i10 = parseInt(i16, 16); // 3
// Calculate the base 10 fraction part
let f10 = parseInt(f16, 16) / Math.pow(16, f16.length); // 0.14158999999999988
// Put the base 10 parts together to find the Number
let b10 = i10 + f10; // 3.14159

# 使用正则和替换函数实现repalce

var string = "heLlo, woRlD!";
string = string.replace( /([a-zA-Z])([a-zA-Z]+)/g, function (match, g1, g2) {
  return g1.toUpperCase() + g2.toLowerCase();
});
console.log( string ); // "Hello, World!"
replaceFunction ParameterMeaning
matchthe substring that matches the entire regular expressiong
g1, g2, g3, ...the matching groups in the regular expression
offsetthe offset of the match in the entire string
stringthe entire string

# Repeat a String

// Version ≥ 6
"abc".repeat(3); // Returns "abcabcabc"
"abc".repeat(0); // Returns ""
"abc".repeat(-1); // Throws a RangeError

// Version < 6
var myString = "abc";
var n = 3;
new Array(n + 1).join(myString); // Returns "abcabcabc"

# Evaluate a callback only if it is passed

function myMethod (cb) {
  // 这样写很普通
  if (cb) {
    cb();
  }
  
  // 这样逼格比较高
  cb && cb();
  // 然而没有类型检查，还是low
}
